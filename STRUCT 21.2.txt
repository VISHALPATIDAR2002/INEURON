#include <stdio.h>
#include <string.h>
struct Employee 
{
    int id;
    char name[20];
    float salary ;
};
struct Employee input() 
{
    struct Employee e;
    printf("Enter employee name ");
    fflush(stdin);
    fgets(e.name,20,stdin);
    e.name[strlen(e.name)-1]='\0';
    printf("Enter employee id ");
    scanf("%d",&e.id);
    printf("Enter employee salary ");
    scanf("%f",&e.salary);
    
} 
int main()
{
    struct Employee e;
    input();
    
}
